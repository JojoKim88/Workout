# convert dataframe columns to snake case
def to_snake_case(column):
    return (
        column.strip()
        .replace(" ", "_")
        .replace(".", "")
        .replace("-", "_")
        .lower()
    )

# Rename columns using the function
data.rename(columns=lambda col: to_snake_case(col), inplace=True)
